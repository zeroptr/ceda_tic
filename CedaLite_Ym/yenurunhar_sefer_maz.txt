select
(Select FAT_IRS_ID from fat_irs where FAT_IRS_SEP=1 and urunhar.fatura_id=fat_irs_id) fat_irs_id,
urunhar_id ,TIP,cast ('URUNHAR SEFER_ID sefer sefer_id alanýnda yok' as varchar(50)) from urunhar
where
SEFER_ID>0
and
SEFER_ID not IN (select sefer_id from sefer where sefer_id=urunhar.sefer_id)

union all

select
(Select FAT_IRS_ID from fat_irs where FAT_IRS_SEP=1 and urunhar.fatura_id=fat_irs_id) fat_irs_id,
urunhar_id ,TIP,cast ('URUNHAR SEFER_BEKLEME sefer sefer_id alanýnda yok' as varchar(50)) from urunhar
where
SEFER_BEKLEME>0
and
SEFER_BEKLEME not IN (select sefer_id from sefer where sefer_id=urunhar.SEFER_BEKLEME)

union all

select
(Select FAT_IRS_ID from fat_irs where FAT_IRS_SEP=1 and urunhar.fatura_id=fat_irs_id) fat_irs_id,
urunhar_id ,TIP,cast ('URUNHAR MAZOT_FISI_ID MAZOT_FISI MAZOT_FIS_ID alanýnda yok' as varchar(50)) from urunhar
where
MAZOT_FISI_ID>0
and
MAZOT_FISI_ID not IN (select MAZOT_FIS_ID from MAZOT_FISI where MAZOT_FIS_ID=urunhar.MAZOT_FISI_ID)
